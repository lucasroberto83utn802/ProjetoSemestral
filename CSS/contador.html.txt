    <script>
        function atualizarContador() {
            const dataAlvo = new Date('2024-06-27T00:00:00');
            const agora = new Date();

            const diferenca = dataAlvo - agora;

            if (diferenca <= 0) {
                document.getElementById('contador').innerText = "A data chegou!";
                clearInterval(interval);
                return;
            }

            const dias = Math.floor(diferenca / (1000 * 60 * 60 * 24));
            const horas = Math.floor((diferenca % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutos = Math.floor((diferenca % (1000 * 60 * 60)) / (1000 * 60));
            const segundos = Math.floor((diferenca % (1000 * 60)) / 1000);

            document.getElementById('contador').innerText = `${dias} dias ${horas} horas ${minutos} minutos ${segundos} segundos restantes atÃ© 27 de Junho de 2024`;
        }

        const interval = setInterval(atualizarContador, 1000);
    </script>
